/*20203152 최정민 구간*/
#include<iostream>
using namespace std;
void check(int data[], int s)
{
    int sum = 0;
    int tmp = 0;
    int cnt = 0;
    bool chk = 0;
    int cases;
    cases = s;
    int n;
    tmp = data[0];
    for (int i = 1; i < cases; i++)
    {
        n= data[i];
        if ( n>= tmp)
        {
            sum = sum+ tmp;
            tmp = n;
            chk = 1;
        }
        else
        {
            if (chk)
            {
                cnt++;
                sum += tmp;
                tmp = n;
            }
            else
                tmp = n;
            chk = 0;
        }
    }
    if (chk)
    {
        sum += tmp;
        cnt++;
    }
    cout << cnt << " " << sum << endl;

}
const int MAX_SIZE = 100;
int main()
{
    int numTestCases;
    int data[MAX_SIZE], size;
    cin >> numTestCases;
    for (int i = 0; i < numTestCases; i++)
    {
        cin >> size;
        for (int j = 0; j < size; j++)
            cin >> data[j];
        check(data, size);
    }
    return 0;
}
