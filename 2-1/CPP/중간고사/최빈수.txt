/*20203152 최정민 최빈수*/
#include<iostream>
using namespace std;
void check()
{
    int N[100] = { 0 };
    int cases;
    int size = 0;
    int ans = 0;
    
    while (1)
    {
        int n;
        cin >> n;
        if (n == 0) break;
        if (n >= 1 && n <= 100)
        {
            N[n]++;
        }
    }
    
    int max = 0;
    int cnt = 0;
    int tmp = 0;
    for (int i = 0; i < 100; i++)
    {
        if (N[i] >= max)
        {
            max = N[i];
            ans = i;
        }
        
    }
    if (max == 0) ans = 0;
    cout << ans << endl;


}
int main()
{
    int cases;
    cin >> cases;
    while (cases--)
    {
        check();
    }

    return 0;
}
