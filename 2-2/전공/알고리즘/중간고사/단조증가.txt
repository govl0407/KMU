int cnt = 0;
int MIN(int a, int b)
{
    int r;
    (a > b) ? r = b : r = a;
    return r;
}
int MAX(int a, int b)
{
    int r;
    (a < b) ? r = b : r = a;
    return r;
}
int rise(int arr[], int len)
{
    int cnt = 1;
    int tmp_cnt = 1;
    for (int i = 1; i < len; i++)
    {
        if (arr[i] >= arr[i - 1])
        {
            tmp_cnt++;
        }
        else
        {
            if (tmp_cnt > cnt)
                cnt = tmp_cnt;
            tmp_cnt = 1;
        }
        if (i == len - 1)
        {
            if (tmp_cnt > cnt)
                cnt = tmp_cnt;
            tmp_cnt = 1; 

        }
    }
    return cnt;
}
int main() {
    int n;
    cin >> n;
    int arr[1001];
    while (n--)
    {
        int num;
        cin >> num;
        cnt = 0;
        for (int i = 0; i < num; i++)
        {
            int k;
            cin >> k;
            arr[i] = k;
        }
        cout << rise(arr, num) << "\n";

}
